class WordPart {
    String syllabary
    String latinPhonetic
}

class Verb extends Word {
    VerbInitialPrefixHolderObject vipho = new VerbInitialPrefixHolderObject()

    //initial prefix
    Affix initialPrefix

    //pronoun prefix
    Affix pronounPrefix

    //reflexive prefix
    Affix reflexivePrefix

    //verb root


    //non final suffix
    Affix nonFinalSuffix
    VerbNonFinalSuffixHolderObject vnfsho = new VerbNonFinalSuffixHolderObject()

    //final suffix
    Affix finalSuffix
    VerbFinalSuffixHolderObject vfsho = new VerbFinalSuffixHolderObject()

}

class Pronominal extends Word {
}

interface AffixI {
    public String toSyllabary(String baseReturnValue, Word word);
    public String toEnglish(String baseReturnValue, Word word);
}

abstract class Affix extends WordPart implements AffixI {

}
